#include<stdio.h>
#include<conio.h>
void main()
{

	int a[6];
	int i,n,f=0;
	printf("Enter Array Elements :\n ");
	for(i=1;i<=5;i++)
	{
		scanf("%d",&a[i]);
	}

	printf("\nArray elements are :\n");

	for(i=1;i<=5;i++)
	{
		printf("%d\t",a[i]);
	}

	printf("Enter Number to Search :\n");
	scanf("%d",&n);

	for(i=1;i<=5;i++)
	{
		if(a[i]==n)
		{
		f=1;
		break;
		}
	}

	if(f==1)
	printf("%d is in the Array\n",n);
	else
	printf("%d is not in the Array\n",n);
	getch();
}
