	void selection(int a[] ,int n )
	{
		int t,min,i,j;
	printf("\nSelection Sort:\n");

	for(i=0;i<=n;i++)
	{
		min=i;
		for(j=i+1;j<=n;j++)
		{
			if(a[min]>a[j])
			{
				min=j;
			}
		
		}
		t=a[min];
		a[min]=a[i];
		a[i]=t;
	}
	}
