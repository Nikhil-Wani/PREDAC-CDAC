#include<stdio.h>
#include<conio.h>
void selection(int[],int);
void main()
{
	int arr[5];
	int i,j,t,n=5;

	for(i=0;i<n;i++)
	{
		printf("Enter the element %d :",i);
		scanf("%d",&arr[i]);
	}

	selection(arr,n);

	printf("The sorted array is:");

	for(i=0;i<n;i++)
		printf("\t%d",arr[i]);
	getch();
}


void selection(int arr[],int n)
{
	int i,j,t=0;

	for(i=0;i<n-1;i++)
	{
		for(j=i+1;j<n;j++)
		{
			if(arr[i]>arr[j])
			{
				t=arr[i];
				arr[i]=arr[j];
				arr[j]=t;
			}
		}
	}
}
